BlankLine :     Sp Newline;
^^Quoted :         '\'' (!'\'' .)* '\''  /    '\"' (!'\"' .)* '\"';
^^HtmlAttribute : (AlphanumericAscii / '-')+ Spnl ('=' Spnl (Quoted / (!'>' Nonspacechar)+)) Spnl ;

Sp :            Spacechar*; 
Spnl :          (Newline / Spacechar)*;
AlphanumericAscii : [A-Za-z0-9] ;

SpecialChar :   '<' / '>' ;
NormalChar :    !( SpecialChar) .;
