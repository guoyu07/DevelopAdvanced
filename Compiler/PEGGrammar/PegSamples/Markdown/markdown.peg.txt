<<Grammar Name = "Markdown"  encoding_class="utf8" >>
^^MarkdownText: S (Image/Link/ InnerCode /Code/Text)+ S expect_file_end ;
^^Image: S '!'Link S;
^^Link: S '['  LinkText ']''('  LinkUrl ')' S ;
^^LinkText: (!SpecialChar .)+	;
^^LinkUrl: (!SpecialChar .)+	;

^^Code: S '```'(![`] .)+ '```' S;
^^InnerCode:S '`' (![`] .)+ '`' S ;

^^Text: S (!SpecialChar .)+ S ;

^SpecialChar :   [~*_`&\[\]()!#] ;

S:        [\n\r\t\v]*	;
expect_file_end:!./ WARNING<" end of file">;
<</Grammar>>
